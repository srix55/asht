                                @Tag(TestMarker.ENDPOINT_TEST)
class KeepaliveResourceTest {
    @Test

        given()
            .baseUri(TestConfig.BASE_URI) // http://localhost:8080
            .basePath(TestConfig.BASE_PATH) // /v1

            .then()

                                RestController Template #

                                @PUT
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Path("/{playerId}/setNameNotes")
public Response setPlayerNameNotes(@HeaderParam("request-id") String requestId, @PathParam("playerId") String playerId, NameNotes nameNotes) {
    String tracerValidity = TracerValidator.isRequestIdValid(requestId);
    if (!tracerValidity.equals(TracerValidator.VALID)) {
        return Response.status(Response.Status.BAD_REQUEST.getStatusCode(),
                Response.Status.BAD_REQUEST.getReasonPhrase() + ": " + tracerValidity).build();
    }

    // TODO Add user id and tenant id
    Tracer tracer = new Tracer(requestId);
    tracer.setTenantId("tenant_brwsky");
    tracer.setUserId("aabuid");

    try {
        tracer.addTraceStart(TASK_SET_NAME_NOTES);
        signage.entities.common.NameNotes eNameNotes = new signage.entities.common.NameNotes(nameNotes.getName(), nameNotes.getNotes(),
                Language.getLanguageByCode(nameNotes.getLanguageCode().getValue()));
        SetPlayerNameNotesUsecaseFactoryProvider.getFactory().make().setPlayerNameNotes(tracer, playerId, eNameNotes);
    } catch (InvalidNameNotesException exception) {
        return ResponseBuilder.build(Response.Status.BAD_REQUEST, exception.getMessage());
    } catch (UnauthorizedException exception) {
        return ResponseBuilder.build(Response.Status.UNAUTHORIZED, "You don't have the necessary permissions to access a player details");
    } catch (NoSuchPlayerException exception) {
        return ResponseBuilder.build(Response.Status.NOT_FOUND, "No such player with player-id: " + playerId);
    } finally {
        tracer.addTraceEndWithMetric(TASK_SET_NAME_NOTES);
    }
    return Response.ok().build();
}
